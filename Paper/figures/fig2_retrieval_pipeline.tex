% fig2_retrieval_pipeline.tex — hybrid recall + RRF + rerank fallback
\tikzset{
  rblock/.style={
    rectangle, draw=black!70, fill=cyan!8, rounded corners=2pt,
    minimum height=0.6cm, minimum width=2.0cm, align=center, font=\scriptsize
  },
  fallback/.style={
    rectangle, draw=black!55, fill=cyan!8, rounded corners=2pt,
    minimum height=0.55cm, minimum width=2.2cm, align=center, font=\scriptsize,
    dashed
  },
  rarrow/.style={-{Stealth[length=2mm]}, semithick, draw=black!65}
}

\begin{tikzpicture}[
  node distance=0.45cm and 0.55cm,
  >=Stealth
]

\node[rblock, minimum width=2.8cm] (query) {Rewritten Query $\tilde{q}$\\[-1pt]\tiny embed $\parallel$ BM25};

\node[rblock, below left=0.9cm and 1.8cm of query, minimum width=2.5cm] (bm25)
  {BM25\\[-1pt]\tiny \texttt{issue}$^2$, \texttt{issueDesc}; fetch=150};
\node[rblock, below=0.9cm of query, minimum width=2.5cm] (knn1)
  {KNN \texttt{text\_vector}\\[-1pt]\tiny fetch=10, $k{=}100$};
\node[rblock, below right=0.9cm and 1.8cm of query, minimum width=2.5cm] (knn2)
  {KNN \texttt{desc\_vector}\\[-1pt]\tiny fetch=10, $k{=}100$};

\node[rblock, below=1.35cm of query, yshift=-0.2cm, minimum width=4.8cm] (rrf)
  {Weighted RRF (\texttt{three\_way})\\[-1pt]$K{=}40$;\; $w_b{=}1.05$, $w_t{=}0.85$, $w_d{=}1.0$\\[-1pt]\tiny top\_n=50};

\node[rblock, below=of rrf, minimum width=3.6cm] (rerank)
  {Reranker (optional)\\[-1pt]\tiny 50 $\to$ top\_k=10;\; timeout 5s};
\node[rblock, below=of rerank, minimum width=3.4cm] (hits)
  {Ranked Hits $\mathcal{C}_{\text{faq}}$\\[-1pt]\tiny top\_k=10};

\node[fallback, right=1.0cm of rerank, minimum width=2.4cm] (fb) {RRF fallback\\[-1pt]\scriptsize config / API error};

\draw[rarrow] (query.south west) -| (bm25.north);
\draw[rarrow] (query) -- (knn1);
\draw[rarrow] (query.south east) -| (knn2.north);
\draw[rarrow] (bm25.south) |- ([xshift=-1.4cm]rrf.north) -- (rrf.north);
\draw[rarrow] (knn1) -- (rrf);
\draw[rarrow] (knn2.south) |- ([xshift=1.4cm]rrf.north) -- (rrf.north);
\draw[rarrow] (rrf) -- (rerank);
\draw[rarrow] (rerank) -- node[right, font=\tiny] {success} (hits);
\draw[rarrow, dashed] (rerank.east) -- (fb.west);
\draw[rarrow, dashed] (fb.south) |- ([xshift=2.0cm]hits.east) -- (hits.east);

\end{tikzpicture}
